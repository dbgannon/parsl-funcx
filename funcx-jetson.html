This is the function we will launch on the jetbot. the dataset argument is a numpy array, the num_centers is an integer and dev is a boolean: if true use the gpu, otherwise use the cpu.

def cluster(dataset_np, num_centers, dev):
    import numpy as np
    import torch
    import sys
    import random
    import time
    sys.path.append('/home/jetbot/')
    from kmeans import random_init, compute_codes, update_centers
    if dev:
        device = torch.device('cuda')
    else:
        device = torch.device('cpu')
 
    dataset = torch.from_numpy(dataset_np).to(device)
    t0 = time.time()
    centers = random_init(dataset, num_centers, dev)
    codes = compute_codes(dataset, centers, dev)
    num_iterations = 0
    while True and num_iterations < 60:
        num_iterations += 1
        centers = update_centers(dataset, codes, num_centers, dev)
        new_codes = compute_codes(dataset, centers, dev)
        # Waiting until the clustering stops updating altogether
        # This is too strict in practice
        if torch.equal(codes, new_codes):
            #sys.stdout.write('\n')
            #print('Converged in %d iterations' % num_iterations)
            break
        codes = new_codes
    t1 =  time.time()
    centers = centers.to(torch.device('cpu'))
    codes = codes.to(torch.device('cpu'))
    return centers, codes, num_iterations, t1-t0
